let simulatedTime = {
  hours: 23,
  minutes: 59,
  seconds: 50

const clockEl = document.getElementById('clock');

function formatTime(h, m, s) {
  return `${String(h).padStart(2, '0')}:${String(m).padStart(2, '0')}:${String(s).padStart(2, '0')}`;
}

function updateTime() {
  simulatedTime.seconds++;
  if (simulatedTime.seconds === 60) {
    simulatedTime.seconds = 0;
    simulatedTime.minutes++;
    if (simulatedTime.minutes === 60) {
      simulatedTime.minutes = 0;
      simulatedTime.hours = (simulatedTime.hours + 1) % 24;
    }
  }

  const timeStr = formatTime(simulatedTime.hours, simulatedTime.minutes, simulatedTime.seconds);
  clockEl.textContent = timeStr;

  // âœ¨ Efecto visual Ã©pico cada segundo
  clockEl.classList.add('blink');
  setTimeout(() => {
    clockEl.classList.remove('blink');
  }, 400);

  if (simulatedTime.hours === 0 && simulatedTime.minutes === 0 && simulatedTime.seconds === 0) {
    setTimeout(launchNewYear, 300);
  } else {
    setTimeout(updateTime, 1000);
  }
}

let canvas, ctx;
let fireworks = [];
let hue = 0;
let isFireworksActive = false;

function initFireworks() {
  if (isFireworksActive) return;
  isFireworksActive = true;

  canvas = document.getElementById('fireworksCanvas');
  if (!canvas) return;
  ctx = canvas.getContext('2d');
  
  function resize() {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
  }
  resize();
  window.addEventListener('resize', resize);

  // ğŸ”¥ Lanzar cohetes cada 2 segundos durante 5 minutos (300 segundos = 150 lanzamientos)
  let launchCount = 0;
  const totalLaunches = 150; // 300s / 2s = 150

  const launchInterval = setInterval(() => {
    if (launchCount >= totalLaunches) {
      clearInterval(launchInterval);
      return;
    }

    // Lanzar 1 o 2 cohetes
    const numRockets = Math.random() > 0.4 ? 1 : 2;
    for (let i = 0; i < numRockets; i++) {
      setTimeout(() => {
        createFirework();
      }, i * 300); // pequeÃ±o delay entre cohetes
    }

    launchCount++;
  }, 2000); // cada 2 segundos

  // AnimaciÃ³n principal
  function animate() {
    if (!isFireworksActive) return;
    requestAnimationFrame(animate);
    
    // Fundido muy suave (para no borrar partÃ­culas rÃ¡pido)
    ctx.fillStyle = 'rgba(0, 0, 0, 0.08)';
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    fireworks = fireworks.filter(fw => {
      fw.update();
      fw.draw(ctx);
      return !fw.isDead;
    });
  }

  animate();
}

class Firework {
  constructor() {
    this.x = 50 + Math.random() * (canvas.width - 100);
    this.y = canvas.height + 20;
    this.vy = -10 - Math.random() * 5; // mÃ¡s velocidad
    this.ay = 0.1; // desaceleraciÃ³n suave
    this.hue = hue;
    hue = (hue + 50) % 360;
    this.trail = [];
    this.maxTrail = 15;
    this.exploded = false;
    this.isDead = false;
    this.particles = [];
    
    // Altura de explosiÃ³n: 70% desde abajo â†’ 30% desde arriba
    this.explodeAtY = canvas.height * 0.3;
  }

  update() {
    if (this.exploded) {
      this.particles.forEach(p => p.update());
      this.particles = this.particles.filter(p => p.life > 0);
      if (this.particles.length === 0) {
        this.isDead = true;
      }
    } else {
      this.vy += this.ay;
      this.y += this.vy;
      
      this.trail.push({x: this.x, y: this.y});
      if (this.trail.length > this.maxTrail) {
        this.trail.shift();
      }
      
      // Â¡Explotar a 70% de la pantalla!
      if (this.y <= this.explodeAtY) {
        this.explode();
      }
    }
  }

  explode() {
    this.exploded = true;
    const particleCount = 300 + Math.random() * 100; // Â¡300-400 partÃ­culas!
    for (let i = 0; i < particleCount; i++) {
      this.particles.push(new Particle(this.x, this.y, this.hue));
    }
  }

  draw(ctx) {
    if (this.exploded) {
      this.particles.forEach(p => p.draw(ctx));
    } else {
      // Rastro mÃ¡s brillante
      for (let i = 0; i < this.trail.length; i++) {
        const point = this.trail[i];
        const alpha = i / this.trail.length;
        const radius = alpha * 3 + 1;
        ctx.beginPath();
        ctx.arc(point.x, point.y, radius, 0, Math.PI * 2);
        ctx.fillStyle = `hsla(${this.hue}, 100%, ${80 - alpha * 30}%, ${alpha * 0.8})`;
        ctx.fill();
      }
      
      // Cabeza del cohete: Â¡mÃ¡s brillante!
      ctx.beginPath();
      ctx.arc(this.x, this.y, 4, 0, Math.PI * 2);
      ctx.fillStyle = '#ffffff';
      ctx.fill();
      ctx.beginPath();
      ctx.arc(this.x, this.y, 6, 0, Math.PI * 2);
      ctx.fillStyle = `hsla(${this.hue}, 100%, 70%, 0.9)`;
      ctx.fill();
    }
  }
}

class Particle {
  constructor(x, y, hue) {
    this.x = x;
    this.y = y;
    this.hue = hue + (Math.random() - 0.5) * 40;
    this.vx = (Math.random() - 0.5) * 16; // mÃ¡s velocidad
    this.vy = (Math.random() - 0.5) * 16 - 8;
    this.gravity = 0.18;
    this.friction = 0.96;
    this.life = 150;
    this.maxLife = 150;
    this.radius = Math.random() * 2 + 1; // mÃ¡s grande
  }

  update() {
    this.vx *= this.friction;
    this.vy += this.gravity;
    this.x += this.vx;
    this.y += this.vy;
    this.life--;
  }

  draw(ctx) {
    const lifePct = this.life / this.maxLife;
    const alpha = lifePct * lifePct * 0.9; // mÃ¡s brillo al inicio
    const size = this.radius * (0.5 + lifePct * 0.8);
    
    ctx.beginPath();
    ctx.arc(this.x, this.y, size, 0, Math.PI * 2);
    ctx.fillStyle = `hsla(${this.hue}, 100%, 80%, ${alpha})`;
    ctx.fill();
    
    // Â¡Destello extra!
    if (lifePct > 0.8) {
      ctx.beginPath();
      ctx.arc(this.x, this.y, size * 1.8, 0, Math.PI * 2);
      ctx.fillStyle = `hsla(${this.hue}, 100%, 90%, ${alpha * 0.5})`;
      ctx.fill();
    }
  }
}
